** config setup
   #+begin_src sh
   git clone https://github.com/daehwannam/nyxt-config.git
   
   if [ -d /home/$USER/.config/nyxt ]; then
       echo "/home/$USER/.config/nyxt exists already"
   else
       mv nyxt-config /home/$USER/.config/nyxt
   fi
   #+end_src
